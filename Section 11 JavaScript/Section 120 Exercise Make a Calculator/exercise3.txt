let firstNum=prompt("What is first number ?");
let secondNum=prompt("What is second number ?");
firstNumd=Number(firstNum);
secondNumd=Number(secondNum);
let sum=firstNumd+secondNumd;
alert(sum);
let detract=firstNumd-secondNumd;
alert(detract);
let divide=firstNumd/secondNumd;
alert(divide);
let multiply =firstNumd*secondNumd;
alert(multiply);
